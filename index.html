Recursion Templates KEY PATTERNS
# RULE: Every recursive function needs a BASE CASE!

# Factorial
def factorial(n):
    if n <= 1: return 1    # base case
    return n * factorial(n-1)

# Fibonacci
def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)

# Sum of digits
def digit_sum(n):
    if n < 10: return n
    return n % 10 + digit_sum(n // 10)

# Power function
def power(base, exp):
    if exp == 0: return 1
    return base * power(base, exp-1)

# Binary search (recursive)
def binary_search(arr, target, lo, hi):
    if lo > hi: return -1
    mid = (lo + hi) // 2
    if arr[mid] == target: return mid
    elif arr[mid] < target: return binary_search(arr, target, mid+1, hi)
    else: return binary_search(arr, target, lo, mid-1)

Pattern 1: Sliding Window HIGH PRIORITY

Use when asked about subarrays/substrings of fixed or variable size.

# Max sum subarray of size k
def max_sum_subarray(arr, k):
    n = len(arr)
    window_sum = sum(arr[:k])
    max_sum = window_sum

    for i in range(k, n):
        window_sum += arr[i] - arr[i-k]  # slide window
        max_sum = max(max_sum, window_sum)

    return max_sum

# Test: max_sum_subarray([2,1,5,1,3,2], 3) → 9

# Longest substring without repeating characters
def longest_unique_substr(s):
    seen = {}
    left = max_len = 0

    for right, ch in enumerate(s):
        if ch in seen and seen[ch] >= left:
            left = seen[ch] + 1
        seen[ch] = right
        max_len = max(max_len, right - left + 1)

    return max_len
Pattern 2: Two Pointers HIGH PRIORITY
# Two sum in SORTED array
def two_sum_sorted(arr, target):
    left, right = 0, len(arr) - 1
    while left < right:
        curr = arr[left] + arr[right]
        if curr == target: return [left, right]
        elif curr < target: left += 1
        else: right -= 1
    return []

# Remove duplicates from sorted array (in-place)
def remove_duplicates(arr):
    if not arr: return 0
    slow = 0
    for fast in range(1, len(arr)):
        if arr[fast] != arr[slow]:
            slow += 1
            arr[slow] = arr[fast]
    return slow + 1

# Check if string is palindrome
def is_palindrome(s):
    left, right = 0, len(s) - 1
    while left < right:
        if s[left] != s[right]: return False
        left += 1; right -= 1
    return True
Pattern 3: Kadane's Algorithm MUST KNOW

Maximum subarray sum — classic TCS question.

def max_subarray_sum(arr):
    max_so_far = arr[0]
    curr_max = arr[0]

    for i in range(1, len(arr)):
        # Either extend current subarray or start fresh
        curr_max = max(arr[i], curr_max + arr[i])
        max_so_far = max(max_so_far, curr_max)

    return max_so_far

Pattern 4: Hashing / Frequency Count VERY USEFUL
# Two Sum (unsorted) — O(n) using hashmap
def two_sum(nums, target):
    seen = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in seen:
            return [seen[complement], i]
        seen[num] = i

# Find first non-repeating character
def first_unique(s):
    from collections import Counter
    cnt = Counter(s)
    for i, ch in enumerate(s):
        if cnt[ch] == 1: return i
    return -1

# Group anagrams together
def group_anagrams(words):
    from collections import defaultdict
    groups = defaultdict(list)
    for w in words:
        key = "".join(sorted(w))  # sorted chars as key
        groups[key].append(w)
    return list(groups.values())

All 4 Traversals MEMORIZE ALL
# INORDER: Left → Root → Right  (gives sorted order for BST!)
def inorder(root):
    if not root: return []
    return inorder(root.left) + [root.val] + inorder(root.right)

# PREORDER: Root → Left → Right  (used for tree copying)
def preorder(root):
    if not root: return []
    return [root.val] + preorder(root.left) + preorder(root.right)

# POSTORDER: Left → Right → Root  (used for tree deletion)
def postorder(root):
    if not root: return []
    return postorder(root.left) + postorder(root.right) + [root.val]

# LEVEL ORDER (BFS): Level by level using queue
from collections import deque
def level_order(root):
    if not root: return []
    result, queue = [], deque([root])
    while queue:
        level = []
        for _ in range(len(queue)):  # process one level
            node = queue.popleft()
            level.append(node.val)
            if node.left:  queue.append(node.left)
            if node.right: queue.append(node.right)
        result.append(level)
    return result
# Output for above tree: [[1], [2, 3], [4, 5]]
Essential Tree Functions PRACTICE THESE
# Height of tree (max depth)
def height(root):
    if not root: return 0
    return 1 + max(height(root.left), height(root.right))

# Count total nodes
def count_nodes(root):
    if not root: return 0
    return 1 + count_nodes(root.left) + count_nodes(root.right)

# Count leaf nodes (nodes with no children)
def count_leaves(root):
    if not root: return 0
    if not root.left and not root.right: return 1
    return count_leaves(root.left) + count_leaves(root.right)

# Check if two trees are identical
def is_same(p, q):
    if not p and not q: return True
    if not p or not q: return False
    return (p.val == q.val and
            is_same(p.left, q.left) and
            is_same(p.right, q.right))

# Max sum path (root to leaf)
def max_path_sum(root):
    if not root: return 0
    left  = max_path_sum(root.left)
    right = max_path_sum(root.right)
    return root.val + max(left, right)
Binary Search Tree (BST) Operations BONUS
# Search in BST — O(log n) average
def search_bst(root, val):
    if not root or root.val == val: return root
    if val < root.val: return search_bst(root.left, val)
    return search_bst(root.right, val)

# Insert in BST
def insert_bst(root, val):
    if not root: return TreeNode(val)
    if val < root.val: root.left = insert_bst(root.left, val)
    else: root.right = insert_bst(root.right, val)
    return root

# Validate if tree is a valid BST
def is_valid_bst(root, lo=float('-inf'), hi=float('inf')):
    if not root: return True
    if not (lo < root.val < hi): return False
    return (is_valid_bst(root.left, lo, root.val) and
            is_valid_bst(root.right, root.val, hi))

BFS — Breadth First Search MUST KNOW

Use for: shortest path, level-by-level traversal, "minimum steps" problems.

from collections import deque

def bfs(graph, start):
    visited = {start}
    queue = deque([start])
    order = []

    while queue:
        node = queue.popleft()
        order.append(node)

        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

    return order

# Find shortest path (BFS)
def shortest_path(graph, start, end):
    visited = {start}
    queue = deque([(start, [start])])  # (node, path)

    while queue:
        node, path = queue.popleft()
        if node == end: return path

        for nb in graph[node]:
            if nb not in visited:
                visited.add(nb)
                queue.append((nb, path + [nb]))
    return []
DFS — Depth First Search MUST KNOW

Use for: exploring all paths, detecting cycles, connected components.

# DFS Recursive
def dfs(graph, node, visited=None):
    if visited is None: visited = set()
    visited.add(node)
    print(node)
    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs(graph, neighbor, visited)

# Count connected components
def count_components(n, edges):
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)
        graph[v].append(u)

    visited = set()
    count = 0

    def explore(node):
        visited.add(node)
        for nb in graph[node]:
            if nb not in visited:
                explore(nb)

    for i in range(n):
        if i not in visited:
            explore(i)
            count += 1
    return count
Number of Islands (Classic Graph Problem) TCS FAVOURITE
# Grid is a 2D array: '1' = land, '0' = water
def num_islands(grid):
    if not grid: return 0
    rows, cols = len(grid), len(grid[0])
    count = 0

    def dfs(r, c):
        if r < 0 or r >= rows or c < 0 or c >= cols: return
        if grid[r][c] != '1': return
        grid[r][c] = '#'  # mark as visited
        dfs(r+1,c); dfs(r-1,c); dfs(r,c+1); dfs(r,c-1)

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '1':
                dfs(r, c)
                count += 1
    return count

Stack & Queue

Stack LIFO

Last In, First Out. Think of a stack of plates.

# Stack using Python list
stack = []
stack.append(1)  # push
stack.append(2)
stack.append(3)
stack.pop()       # 3 — pop
stack[-1]         # 2 — peek
len(stack) == 0   # is empty?

# Balanced Parentheses
def is_balanced(s):
    stack = []
    pairs = {')':'(', ']':'[', '}':'{'}
    for ch in s:
        if ch in '([{':
            stack.append(ch)
        elif ch in pairs:
            if not stack or stack[-1] != pairs[ch]:
                return False
            stack.pop()
    return len(stack) == 0
Queue FIFO

First In, First Out. Think of a line at a ticket counter.

# Queue using collections.deque
from collections import deque

q = deque()
q.append(1)    # enqueue
q.append(2)
q.append(3)
q.popleft()     # 1 — dequeue
q[0]            # 2 — peek front

# deque also works as a double-ended queue
q.appendleft(0)  # add to front
q.pop()          # remove from back
# Very useful for sliding window problems!
Next Greater Element CLASSIC PROBLEM
# For each element, find the next greater element to its right
# [2, 1, 2, 4, 3] → [4, 2, 4, -1, -1]
def next_greater(arr):
    n = len(arr)
    result = [-1] * n
    stack = []  # stores indices

    for i in range(n):
        # Pop all elements smaller than current
        while stack and arr[stack[-1]] < arr[i]:
            idx = stack.pop()
            result[idx] = arr[i]
        stack.append(i)

    return result
